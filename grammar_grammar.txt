
grammar {
    rule+
}

rule {
    NON_TERMINAL OCURLY rule_function CCURLY
}

terminal {
    TERMINAL_NAME |
    TERMINAL_OPER |
    TERMINAL_SYMBOL
}

rule_element {
    NON_TERMINAL |
    terminal |
    rule_function
}

rule_element_list {
    OPAREN rule_element+ CPAREN |
    rule_element+
}

rule_function {
    or_func |
    zero_or_more_func |
    zero_or_one_func |
    one_or_more_func
}

one_or_more_func {
    rule_element_list ONE_OR_MORE
}

zero_or_one_func {
    rule_element_list ZERO_OR_ONE
}

zero_or_more_func {
    rule_element_list ZERO_OR_MORE
}

or_func {
    rule_element PIPE rule_element
}
