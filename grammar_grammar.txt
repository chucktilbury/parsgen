# This is the grammar for the input of the parser template generator.

grammar {
    +rule END_OF_INPUT
}

rule {
    NON_TERMINAL '{' +rule_element '}'
}

rule_element {
    NON_TERMINAL |
    TERMINAL_NAME |
    TERMINAL_OPER |
    TERMINAL_SYMBOL |
    or_func |
    zero_or_more_func |
    zero_or_one_func |
    one_or_more_func |
    group_func
}

one_or_more_func {
    '+' rule_element
}

zero_or_one_func {
    '?' rule_element
}

zero_or_more_func {
    '*' rule_element
}

or_func {
    '|' rule_element
}

group_func {
    '(' +rule_element ')'
}
