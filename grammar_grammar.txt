
grammar {
    +rule END_OF_INPUT
}

rule {
    NON_TERMINAL OCURLY rule_element_list CCURLY
}

terminal {
    TERMINAL_NAME |
    TERMINAL_OPER |
    TERMINAL_SYMBOL
}

rule_element {
    NON_TERMINAL |
    terminal |
    rule_function
}

rule_element_list {
    OPAREN +rule_element CPAREN |
    +rule_element
}

rule_function {
    or_func |
    zero_or_more_func |
    zero_or_one_func |
    one_or_more_func
}

one_or_more_func {
    ONE_OR_MORE rule_element_list
}

zero_or_one_func {
    ZERO_OR_ONE rule_element_list
}

zero_or_more_func {
    ZERO_OR_MORE rule_element_list
}

or_func {
    PIPE rule_element_list
}
