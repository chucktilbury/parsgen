
grammar {
    rule+
}

rule {
    NON_TERMINAL OCURLY rule_function+ CCURLY
}

terminal {
    TERMINAL_NAME |
    TERMINAL_OPER |
    TERMINAL_SYMBOL
}

rule_element {
    NON_TERMINAL |
    terminal
}

rule_element_list {
    OPAREN rule_element+ CPAREN |
    rule_element+
}

rule_function {
    or_func |
    zero_or_more_func |
    zero_or_one_func |
    one_or_more_func |
    rule_element
}

one_or_more_func {
    OPAREN rule_function+ CPAREN ONE_OR_MORE |
    rule_element ONE_OR_MORE
}

zero_or_one_func {
    OPAREN rule_function+ CPAREN ZERO_OR_ONE |
    rule_element ZERO_OR_ONE
}

zero_or_more_func {
    OPAREN rule_function+ CPAREN ZERO_OR_MORE |
    rule_element ZERO_OR_MORE
}

or_func {
    rule_element_list PIPE rule_element_list
}
